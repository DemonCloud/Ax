var model = new aix.model.extend({
	data:"hello",
	events:{
		change:function(val){
			this.view.dispatch("upinput",null,[val]);
		}
	}
});

var view = new aix.view({
	el:"#render",

	template:_.doom(z("#tmp").html()),
	render:function(data){
		z(this.el).append(this.template(data));
	},

	events:{
		init:function(){
			z(this.el).find("#ri")
								 .value("Hello Aix")
			this.render({text:"Hello Aix"});

			this.model = new model({
				view : this
			});
		},

		upinput:function(txt){
			z(this.el).find("#rv").fill(txt||"");
		},

		"input:#ri":function(event){
			var view = event.data.self;
			view.model.set(_.encodeHTML(z(this).value()));
		}
	},
});
<div id="rv">{{-text}}</div>
